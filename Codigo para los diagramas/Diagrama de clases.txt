class Hotel {
    -nombre : String
    -direccion : String
    +obtenerHabitacionesDisponibles(tipo : TipoHabitacion)
    +obtenerPrecioHabitacion(tipo : TipoHabitacion)
    +cambiarPrecio(tipo : TipoHabitacion, precio : double)
    +calcularGanancias(mes : int)
}

class Habitacion {
    -numero : int
    -tipo : TipoHabitacion
    -precio : double
    -disponible : boolean
    +obtenerPrecio()
    +mostrarFoto()
}

enum TipoHabitacion {
    SIMPLE
    DOBLE
    MATRIMONIAL
}

class Cliente {
    -idCliente : int
    -nombre : String
    -tipo : TipoCliente
    +obtenerDescuento()
}

enum TipoCliente {
    HABITUAL
    ESPORADICO
}

class Reserva {
    -numeroReserva : int
    -fechaInicio : Date
    -cantidadNoches : int
    +calcularTotal()
    +crearReserva()
    +cancelarReserva()
}
