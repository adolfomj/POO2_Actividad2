
@startuml

actor Recepcionista
participant Sistema
participant Habitacion
participant Reserva
participant Cliente

Recepcionista -> Sistema : solicitarReserva(tipoHabitacion, cliente, noches)

Sistema -> Habitacion : buscarDisponible(tipoHabitacion)
Habitacion --> Sistema : habitacionDisponible

Sistema -> Reserva : crearReserva(cliente, habitacion, noches)

Reserva -> Cliente : obtenerDescuento()
Cliente --> Reserva : descuento

Reserva -> Habitacion : obtenerPrecio()
Habitacion --> Reserva : precio

Reserva -> Reserva : calcularTotal()

Reserva --> Sistema : total

Sistema --> Recepcionista : confirmarReserva

@enduml
